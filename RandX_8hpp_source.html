816 // ── A3 跨平台 OS 密码学熵源 ──
817 // 用 OS 密码学 API 填充 [buf, buf+n) 字节；成功返回 true。
818 // 平台优先级：Windows BCryptGenRandom → Linux getrandom → macOS SecRandomCopyBytes → std::random_device 兜底
819 // 注：getrandom 可能短读，内部循环直至填满；BCryptGenRandom/SecRandomCopyBytes 一次填满
820 [[nodiscard]]
821 inline bool GetOsEntropyBytes(void* buf, std::size_t n) noexcept
822 {
823 if (n == 0) return true;
824 auto* p = static_cast<std::uint8_t*>(buf);
825
826# if defined(_WIN32) && __has_include(<bcrypt.h>)
827 // Windows: BCryptGenRandom（一次调用填满）
828 // NTSTATUS >= 0 即 NT_SUCCESS（不能 == 0，正向 informational code 也属成功）
829 return (::BCryptGenRandom(nullptr, p, static_cast<ULONG>(n),
830 BCRYPT_USE_SYSTEM_PREFERRED_RNG) >= 0);
831
832# elif defined(__linux__) && __has_include(<sys/random.h>)
833 // Linux: getrandom（循环处理短读与 EINTR）
834 std::size_t filled = 0;
835 while (filled < n)
836 {
837 const ssize_t ret = ::getrandom(p + filled, n - filled, 0);
838 if (ret < 0)
839 {
840 if (errno == EINTR) continue; // 被信号打断，重试
841 return false; // ENOSYS/EFAULT 等不可恢复错误
842 }
843 if (ret == 0) return false;
844 filled += static_cast<std::size_t>(ret);
845 }
846 return true;
847
848# elif defined(__APPLE__) && __has_include(<Security/Security.h>)
849 // macOS: SecRandomCopyBytes（一次调用填满）
850 return (::SecRandomCopyBytes(kSecRandomDefault, n, p) == errSecSuccess);
851
852# else
853 // 回退：std::random_device（标准未保证密码学安全）
854 try
855 {
856 std::random_device rd;
857 std::size_t filled = 0;
858 while (filled < n)
859 {
860 const auto r = rd();
861 const std::size_t chunk = (std::min)(sizeof(r), n - filled);
862 std::memcpy(p + filled, &r, chunk);
863 filled += chunk;
864 }
865 return true;
866 }
867 catch (...)
868 {
869 return false;
870 }
871# endif
872 }

895 // ChaCha20 quarter-round（仅 add/xor/rotl，常时间友好）
896 static void ChaCha20QuarterRound(std::uint32_t& a, std::uint32_t& b,
897 std::uint32_t& c, std::uint32_t& d) noexcept
898 {
899 a += b; d ^= a; d = RotL(d, 16);

978 {

989 // 若解析失败（读取不足或流错误），setstate(failbit) 且引擎状态保持不变
990 // （与 std::random_engine 一致：先读取到临时 state，全部成功才 deserialize）
991 template <class CharT, class Traits, detail::SerializableEngine Engine>
992 std::basic_istream<CharT, Traits>&
993 operator>>(std::basic_istream<CharT, Traits>& is, Engine& engine)
994 {
995 typename Engine::state_type state{};
996 std::size_t i = 0;
997 for (; i < state.size() && is; ++i)
998 is >> state[i];
999
1000 if (i == state.size() && is)
1001 {
1002 engine.deserialize(state);
1003 }
1004 else
1005 {
1006 is.setstate(std::ios_base::failbit);
1007 }
1008 return is;
1009 }

1731 // ChaCha20 (RFC 7539) — CSPRNG 引擎实现
1732 //
1733 // 状态矩阵布局（16 × uint32，常数省略存于 m_state[0..11]）：
1734 // 0 1 2 3 "expa" "nd 3" "2-by" "te k" ← 常数（generateBlock 时补齐）
1735 // 4 5 6 7 key[0] key[1] key[2] key[3] ← m_state[0..3]
1736 // 8 9 10 11 key[4] key[5] key[6] key[7] ← m_state[4..7]
1737 // 12 13 14 15 ctr nonce[0] nonce[1] nonce[2]← m_state[8..11]
1738 //
1739 // 生成流程：operator() → reseedIfNecessary → (缓存耗尽时)generateBlock → 取 8 字节
1740 //
1741
1742 // 构造方式 1：从 OS 熵自动播种（密码学安全，默认）
1743 inline ChaCha20::ChaCha20()
1744 : m_state{}, m_buffer{}, m_bufferPos(64), m_bytesSinceReseed(0)
1745 {
1746 reseed(); // 从 OS 熵获取 key + nonce，重置 counter
1747 }
1748
1749 // 构造方式 2：显式种子（仅测试/复现，非密码学安全）
1750 // 用 SplitMix64 将 64-bit 种子扩展为 32 字节 key + 12 字节 nonce
1751 inline ChaCha20::ChaCha20(const std::uint64_t seed)
1752 : m_state{}, m_buffer{}, m_bufferPos(64), m_bytesSinceReseed(0)
1753 {
1754 SplitMix64 sm{ seed };
1755 // key: 前 4 次 SplitMix64 输出，每次 8 字节按小端序拆为 2 个 uint32
1756 for (int i = 0; i < 4; ++i)
1757 {
1758 const std::uint64_t v = sm();
1759 m_state[i * 2] = static_cast<std::uint32_t>(v);
1760 m_state[i * 2 + 1] = static_cast<std::uint32_t>(v >> 32);
1761 }
1762 // nonce: 第 5 次输出（8 字节）+ 第 6 次输出低 4 字节（丢弃高 4 字节）
1763 {
1764 const std::uint64_t v5 = sm();
1765 m_state[9] = static_cast<std::uint32_t>(v5);
1766 m_state[10] = static_cast<std::uint32_t>(v5 >> 32);
1767 }
1768 m_state[11] = static_cast<std::uint32_t>(sm());
1769 m_state[8] = 0; // counter 初值 = 0
1770 }

1772 // 构造方式 3：直接指定 key + nonce + counter
1773 inline ChaCha20::ChaCha20(const std::uint8_t* key, std::size_t keyLen,
1774 const std::uint8_t* nonce, std::size_t nonceLen,
1775 const std::uint32_t counter)
1776 : m_state{}, m_buffer{}, m_bufferPos(64), m_bytesSinceReseed(0)
1777 {
1778 if (keyLen != 32)
1779 throw std::invalid_argument("ChaCha20: key must be 32 bytes");
1780 if (nonceLen != 12)
1781 throw std::invalid_argument("ChaCha20: nonce must be 12 bytes");
1782 // key → m_state[0..7]（小端序）
1783 for (int i = 0; i < 8; ++i)
1784 {
1785 m_state[i] = static_cast<std::uint32_t>(key[i * 4])
1786 | (static_cast<std::uint32_t>(key[i * 4 + 1]) << 8)
1787 | (static_cast<std::uint32_t>(key[i * 4 + 2]) << 16)

1801 // 生成一个 ChaCha20 block（64 字节）填充 m_buffer
1802 inline void ChaCha20::generateBlock()
1803 {
1804 // 构造完整 16-word 状态：常数 + key + counter + nonce
1805 std::array<std::uint32_t, 16> state{};
1806 state[0] = detail::ChaCha20Constants[0];
1807 state[1] = detail::ChaCha20Constants[1];
1808 state[2] = detail::ChaCha20Constants[2];
1809 state[3] = detail::ChaCha20Constants[3];
1810 for (int i = 0; i < 8; ++i) state[4 + i] = m_state[i]; // key
1811 state[12] = m_state[8]; // counter
1812 state[13] = m_state[9]; // nonce[0]
1813 state[14] = m_state[10]; // nonce[1]
1814 state[15] = m_state[11]; // nonce[2]
1815
1816 std::array<std::uint32_t, 16> working = state;
1817
1818 // 20 轮 = 10 次 double-round（列轮 + 对角轮）
1819 for (int i = 0; i < 10; ++i)
1820 {
1821 // 列轮 QR 顺序：(0,4,8,12) (1,5,9,13) (2,6,10,14) (3,7,11,15)
1822 detail::ChaCha20QuarterRound(working[0], working[4], working[8], working[12]);
1823 detail::ChaCha20QuarterRound(working[1], working[5], working[9], working[13]);
1824 detail::ChaCha20QuarterRound(working[2], working[6], working[10], working[14]);
1825 detail::ChaCha20QuarterRound(working[3], working[7], working[11], working[15]);
1826 // 对角轮 QR 顺序：(0,5,10,15) (1,6,11,12) (2,7,8,13) (3,4,9,14)
1827 detail::ChaCha20QuarterRound(working[0], working[5], working[10], working[15]);
1828 detail::ChaCha20QuarterRound(working[1], working[6], working[11], working[12]);
1829 detail::ChaCha20QuarterRound(working[2], working[7], working[8], working[13]);
1830 detail::ChaCha20QuarterRound(working[3], working[4], working[9], working[14]);
1831 }
1832
1833 // 加初始状态后按小端序输出 64 字节到 m_buffer
1834 for (int i = 0; i < 16; ++i)
1835 {
1836 const std::uint32_t v = working[i] + state[i];
1837 m_buffer[i * 4 + 0] = static_cast<std::uint8_t>(v);
1838 m_buffer[i * 4 + 1] = static_cast<std::uint8_t>(v >> 8);
1839 m_buffer[i * 4 + 2] = static_cast<std::uint8_t>(v >> 16);
1840 m_buffer[i * 4 + 3] = static_cast<std::uint8_t>(v >> 24);
1841 }
1842
1843 ++m_state[8]; // 递增 counter（2^20 字节阈值远早于 2^32 回绕，自动 reseed 防止复用）
1844 m_bufferPos = 0;
1845 }

1855 inline void ChaCha20::reseed()
1856 {
1857 std::array<std::uint8_t, 44> seed; // 32(key) + 12(nonce)
1858 SecureRandomBytes(seed.data(), seed.size());
1859 // key → m_state[0..7]（小端序）
1860 for (int i = 0; i < 8; ++i)
1861 {
1862 m_state[i] = static_cast<std::uint32_t>(seed[i * 4])
1863 | (static_cast<std::uint32_t>(seed[i * 4 + 1]) << 8)
1864 | (static_cast<std::uint32_t>(seed[i * 4 + 2]) << 16)
1865 | (static_cast<std::uint32_t>(seed[i * 4 + 3]) << 24);
1866 }
1867 // nonce → m_state[9..11]（小端序）
1868 for (int i = 0; i < 3; ++i)
1869 {
1870 m_state[9 + i] = static_cast<std::uint32_t>(seed[32 + i * 4])
1871 | (static_cast<std::uint32_t>(seed[32 + i * 4 + 1]) << 8)
1872 | (static_cast<std::uint32_t>(seed[32 + i * 4 + 2]) << 16)
1873 | (static_cast<std::uint32_t>(seed[32 + i * 4 + 3]) << 24);
1874 }
1875 m_state[8] = 0; // counter 重置
1876 m_bufferPos = 64; // 强制下次 operator() 触发新 block
1877 m_bytesSinceReseed = 0;
1878 detail::SecureWipe(seed.data(), seed.size()); // 擦除栈上密钥材料
1879 detail::SecureWipe(m_buffer.data(), m_buffer.size()); // 擦除旧 keystream
1880 }

2489 // RandSample 迭代器版（v1.2 新增）
2490 // 路径 1：随机访问迭代器 —— hash-set / 索引数组双分支

2499 // 路径 1：随机访问迭代器（hash-set / 索引数组双分支）
2500 template <std::random_access_iterator It>
2501 [[nodiscard]]
2502 inline std::vector<std::iter_value_t<It>>
2503 RandSample(It first, It last, std::iter_difference_t<It> n)
2504 {
2505 using Diff = std::iter_difference_t<It>;
2506 using T = std::iter_value_t<It>;
2507 const Diff size = std::distance(first, last);
2508 if (n <= 0 || size == 0)
2509 return {};
2510 if (n >= size)
2511 return std::vector<T>(first, last);
2512
2513 auto& rng = DefaultEngine();
2514
2515 // 分支选择：n·K < size 时 hash-set 内存优（O(n)）；否则索引数组常数优（O(N)）
2516 // 用 uint64_t 避免 n*K 溢出（n 是 iter_difference_t，可能 32 位）
2517 const auto sizeU = static_cast<std::uint64_t>(size);
2518 // 线性阈值：n·K < size 时用 hash-set（实测交叉点 n≈N/127，K=64 留 2× 裕度）
2519 if (static_cast<std::uint64_t>(n) * detail::HashSetThresholdK < sizeU)
2520 {
2521 // hash-set 分支：O(n) 内存，O(n) 期望时间
2522 std::unordered_set<Diff> selected;
2523 selected.reserve(static_cast<std::size_t>(n));
2524 std::vector<T> result;
2525 result.reserve(static_cast<std::size_t>(n));
2526 while (result.size() < static_cast<std::size_t>(n))
2527 {
2528 std::uniform_int_distribution<Diff> dist(Diff{0}, static_cast<Diff>(sizeU - 1));
2529 const Diff idx = dist(rng);
2530 if (selected.insert(idx).second)
2531 result.push_back(first[idx]);
2532 }
2533 return result;
2534 }
2535
2536 // 索引数组分支：O(N) 内存，O(N) 时间，无碰撞
2537 std::vector<Diff> indices(static_cast<std::size_t>(size));
2538 for (Diff i = 0; i < size; ++i)
2539 indices[static_cast<std::size_t>(i)] = i;
2540
2541 // Fisher-Yates 前 n 步：j ∈ [i, size-1]
2542 for (Diff i = 0; i < n; ++i)
2543 {
2544 std::uniform_int_distribution<Diff> dist(i, static_cast<Diff>(size - 1));
2545 const Diff j = dist(rng);
2546 std::swap(indices[static_cast<std::size_t>(i)],
2547 indices[static_cast<std::size_t>(j)]);
2548 }
2549
2550 std::vector<T> result;
2551 result.reserve(static_cast<std::size_t>(n));
2552 for (Diff i = 0; i < n; ++i)
2553 result.push_back(first[indices[static_cast<std::size_t>(i)]]);
2554 return result;
2555 }

2562 // 路径 2：输入迭代器（reservoir sampling, Algorithm R, i+1 修复）
2563 template <std::input_iterator It>
2564 requires (!std::random_access_iterator<It>)
2565 [[nodiscard]]
2566 inline std::vector<std::iter_value_t<It>>
2567 RandSample(It first, It last, std::iter_difference_t<It> n)
2568 {
2569 using Diff = std::iter_difference_t<It>;
2570 using T = std::iter_value_t<It>;
2571 if (n <= 0)
2572 return {};
2573
2574 std::vector<T> reservoir;
2575 reservoir.reserve(static_cast<std::size_t>(n));
2576
2577 // 填满蓄水池
2578 Diff i = 0;
2579 for (; i < n && first != last; ++i, ++first)
2580 reservoir.push_back(*first);
2581
2582 if (first == last)
2583 return reservoir; // 元素不足 n，返回全部
2584
2585 // Algorithm R：第 i 个元素（i >= n，0-indexed）以 n/(i+1) 概率替换蓄水池随机位置
2586 // 关键：j ∈ [0, i]（闭区间），uniform_int_distribution(0, i) 正好是 [0, i] 闭区间
2587 auto& rng = DefaultEngine();

2589 {
2590 std::uniform_int_distribution<Diff> dist(Diff{0}, i);
2591 const Diff j = dist(rng);
2592 if (j < n)
2593 reservoir[static_cast<std::size_t>(j)] = *first;
2594 }
2595 return reservoir;
